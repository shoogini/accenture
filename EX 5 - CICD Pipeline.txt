2.
app.py

@'
def add(a, b):
    return a + b

if __name__ == "__main__":
    print("Hello from Python app")
    print("2 + 3 =", add(2, 3))
'@ | Out-File -Encoding UTF8 app.py

# create a test
mkdir tests
@'
from app import add

def test_add():
    assert add(2, 3) == 5
'@ | Out-File -Encoding UTF8 tests\test_app.py

# create requirements
@'pytest==8.3.2'@ | Out-File -Encoding UTF8 requirements.txt

# create .gitignore
@'
__pycache__/
.venv/
.pytest_cache/
*.pyc
'@ | Out-File -Encoding UTF8 .gitignore

Workflows

file ---  .github/workflows/python.yml


name: CI
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: python-version: "3.11"
      - run: python -m pip install --upgrade pip
      - run: pip install -r requirements.txt
      - run: pytest -q
